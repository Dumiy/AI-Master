#deterministic function for output with sigmoid function as output 
@mc.deterministic
def out_2(w1=w1,b1=b1,w2=w2,b2=b2,x=X):
    out_1 = np.tanh(np.add(np.matmul(x,w1),b1))
    out_1 = np.add(np.matmul(out_1,w2),b2)
    out = 1/(1 + np.exp(-(out_1)))[:,0]
    return out
# trainign with bernouli output
bern = mc.Bernoulli("out",out_2,value=Y,observed=True)

class Network(nn.Module):
    def __init__(self):
        super(Network, self).__init__()
        self.fc1 = nn.Linear(2,2)
        self.fc2 = nn.Linear(2,1)
    def forward(self,x):
        x = F.tanh(self.fc1(x))
        x = F.sigmoid(self.fc2(x))
        return x.squeeze(1)

class TrainLoader(Dataset):
    def __init__(self,data,labels):
        self.data = data
        self.labels = labels
    def __len__(self):
        return self.data.shape[0]
    def __getitem__(self, idx):
        sample = self.data[idx]
        label = self.labels[idx]
        return torch.tensor(sample),torch.tensor(label)

        data = data.cuda().float()

        #print(target)

        #print(outputs)
        loss = criterion(outputs, target.float())
        loss.backward()
        optim.step()

        # print statistics
        running_loss += loss.item()

def test(criterion,test_loader):
    total = 0
    correct = 0
    with torch.no_grad():
        for data,target in test_loader:
            data,target = data.cuda(),target.cuda()
            outputs = model(data.float())
            pred = (outputs > 0.5).type(torch.uint8)
            #print(pred)
            #print(target)
            total += target.shape[0]
            correct += (pred == target).sum().item()
    print('Accuracy of the network on the 10000 test images: %d %%' % (
        100 * correct / total))
